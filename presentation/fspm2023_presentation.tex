\begin{frame}{Normalised Evaluation Metric}

$ \text{NMSE} = \dfrac{1}{N}\dfrac{\sum_{i=0}^{N-1}\left[y-\hat{y}\right]^2}{\text{var}(y)} $

\begin{tabular}{cl}
+ & normalised metric \\
+ & easy to compare different targets \\
- & phase/time sensitive \\
- & sensitive to outliers
\end{tabular}

\end{frame}
